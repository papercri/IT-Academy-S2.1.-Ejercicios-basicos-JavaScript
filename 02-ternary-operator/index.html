Operador ternario básico: Escribe una función puedeConducir que acepte la edad como parámetro y utilice el operador ternario para determinar si el usuario puede conducir. Si la edad es 18 o más, debe devolver 'Puedes conducir'. Si no, debe devolver 'No puedes conducir'. 

    
            function puedeConducir (edad){
                if(Number.isInteger(edad)){
                    edad >= 18 ? console.log("Puedes conducir") : console.log("No puedes conducir");
                } else {
                    console.log("Por favor escriba la edad del conductor");
                }
            };
    
            puedeConducir(10);
            puedeConducir(60);
            puedeConducir(10.8);

Uso enlazado de operadores ternarios: Escribe una expresión que utilice enlaces de operadores ternarios para determinar si un número es positivo, negativo o cero.

    
            function enlaceDeOperadoresTernarios(num) {
                 return isNaN(num) 
                    ? console.log("por favor introduce un numero") 
                    : (num === 0) 
                        ? console.log("El numero es CERO") 
                        : (num > 0) 
                            ? console.log("El numero es POSITIVO") 
                            : console.log("El numero es NEGATIVO");
            }
           
            enlaceDeOperadoresTernarios(10);
            enlaceDeOperadoresTernarios(-10);
            enlaceDeOperadoresTernarios(0);
            enlaceDeOperadoresTernarios("ok");

            function encontrarMaximo(a, b, c) {
                return (isNaN(a) || isNaN(b) || isNaN(c)) 
                    ? console.log("por favor introduce tres numeros") 
                    : ((a == b) && (b == c))
                    ? console.log("Hay tres numeros iguales, pon almenos dos numeros diferentes") 
                    : (a > b && a > c) 
                        ? console.log("A es el numero mayor")
                        : (b > a && b > c) 
                            ? console.log("B es el numero mayor")
                            : console.log("C es el numero mayor");
            }

Operador ternario en un bucle: Escribe una función parOImpar que acepte un array de números y utilice un bucle para recorrer el array. Dentro del bucle, utiliza el operador ternario para determinar si cada número es par o impar.

            function parOImpar(numbers){
                for (let n of numbers) {
                    console.log(!isNaN(n) ? (n % 2 === 0 ? `${n} es un número par` : `${n} es un número impar`) : `${n} no es un número válido`);
                }
            }
    
            parOImpar([2, 7, 8]);
